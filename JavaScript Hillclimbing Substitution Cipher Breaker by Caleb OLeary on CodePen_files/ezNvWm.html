Hillclimbing Demo, pick a random position and find the nearest peak, if at peak, pick another random position.
<div class="target"></div>
.bar {
  width:10px;
  margin-right:3px;
  background:#5050F5;
  float:left;
}
var dataSet = [3,9,15,20,25,46,67,67.5,68,61,54,43,32,22,12,34,56,62.5,69,69.5,70,63.5,57,50,43,37.5,32,22,12,23,34,44,54,39.5,25,17.5,10,6.5,3,3,3,9,15,20,25,46,67,67.5,68,61,54,43,32,22,12,34,56,62.5,69,65,61,59,57,50,43,37.5];
for (var i = 0; i < dataSet.length; i++) {
    $('.target').append('<div class="bar" style="height:'+dataSet[i]*2+'px;margin-top:'+(140-dataSet[i]*2)+'px;"></div>');
}
function search(position){
    $('.bar').css('background', '#5050F5');
    $('.bar').eq(position).css('background', 'red');
    setTimeout(function() {
        if (typeof dataSet[position + 1] !== 'undefined' && dataSet[position + 1] > dataSet[position]) {
            position++;
            search(position);
        }
        else if (typeof dataSet[position - 1] !== 'undefined' && dataSet[position - 1] > dataSet[position]) {
            position--;
            search(position);
        }
        else {
            var random = Math.floor(Math.random() * dataSet.length-1) + 1;
            //peaks.push(dataSet[position]);
            //letsClimb = false;
            search(random);
        }
    }, 500);
}
search(3);
